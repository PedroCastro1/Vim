syntax on 

" Basic Setup 
set nocompatible
set cursorline
set ruler
set number
set showcmd
set wildmenu
set wildmode=longest:list,full
set nobackup
set hidden
set visualbell
set novb

set laststatus=2
set statusline=%F%m%r%h%w%=(%{&ff}/%Y)\ (row\ %l\/%L,\ col\ %c)
set title
set titlestring=%F

" Identation 
set expandtab
set shiftwidth=2
set tabstop=2
set softtabstop=2
set textwidth=100
set smarttab
set showmatch
set showmode
set autoindent
set smartindent

" Maps
noremap <F5> :w <CR>
nnoremap ;; A;<CR>
map <Down> <NOP>
map <Up> <NOP>
map <Left> <NOP>
map <Right> <NOP>
inore jj <Esc>
inoremap {<CR>  <Space>{<CR>}<C-o>O
inoremap ( ()<left>
inoremap [ []<left>
inoremap " ""<left>

" Breaking Lines
nnoremap S i<CR><Esc>^mwgk<CR>`w 
"Au
au FocusLost * :wa
au VimResized * exe "normal! \<c-w>="

augroup line_return
    au!
    au BufReadPost *
        \ if line("'\"") > 0 && line("'\"") <= line("$") |
        \     execute 'normal! g`"zvzz' |
        \ endif
augroup END
